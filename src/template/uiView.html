app.config(function($urlRouterProvider, $stateProvider) {
  $urlRouterProvider.otherwise("/home");
  $stateProvider
    .state("uiView", {
      url: "/uiView",
      templateUrl: "13131"})
    .state("ngForm",{
      url:'/ngFormDemo',
      templateUrl:"src"})
    .state("bootstrap",{
      url:'/bootstrap',
      templateUrl:'src/template/bootstrap.html'})
    .state("home",{
      url:"/home",
      template:"我是首页"
    })
  })
